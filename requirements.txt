SECTION_WEIGHTS = {
    # Strong plaintiff-favoring precedents
    "consumer protection act": +15,
    "section 138":             +20,  # Cheque bounce — very high conviction rate
    "motor accident":          +18,
    "wrongful termination":    +10,
    "rent control":            +12,
    
    # Factors that reduce probability
    "limitation period":       -20,  # Time-barred cases lose almost always
    "arbitration clause":      -15,  # Forced to arbitration, not court
    "force majeure":           -10,
    "as is where is":          -8,
}

COMPLEXITY_PENALTIES = {
    "property dispute":   -12,  # Takes forever, unpredictable
    "cheque bounce":        0,  # Clear cut
    "consumer complaint":  +5,  # Consumer forums are plaintiff-friendly
}

def compute_win_probability(query: str, live_context: str) -> dict:
    query_lower = (query + live_context).lower()
    base_score  = 50  # neutral starting point
    factors     = []

    for keyword, weight in SECTION_WEIGHTS.items():
        if keyword in query_lower:
            base_score += weight
            direction   = "+" if weight > 0 else ""
            factors.append(f"{direction}{weight}% — {keyword} detected")

    # Cap between 15 and 88 (no case is ever certain)
    final = max(15, min(88, base_score))

    return {
        "probability": final,
        "factors":     factors,
        "confidence":  "High" if len(factors) >= 3 else "Medium" if len(factors) >= 1 else "Low",
    }Now when a recruiter asks "how did you compute that?" you have a real answer with explainable factors shown in the UI.Step 1.2 — Real PDF Risk Scoring with NLPReplace 65 + (len(full_text) % 25) with actual clause analysis:python# utils/contract_scorer.py
import re

RED_FLAG_CLAUSES = {
    "unilateral termination":     ("HIGH",   25, "Employer can exit without cause"),
    "non-compete":                ("HIGH",   20, "May be unenforceable under S.27 ICA"),
    "intellectual property":      ("HIGH",   18, "Overbroad IP assignment"),
    "liquidated damages":         ("MEDIUM", 12, "Pre-determined penalties may be excessive"),
    "indemnification":            ("MEDIUM", 10, "Unlimited indemnity is risky"),
    "limitation of liability":    ("MEDIUM",  8, "May cap your recovery unfairly"),
    "governing law":              ("LOW",     5, "Check if jurisdiction is inconvenient"),
    "arbitration":                ("MEDIUM", 10, "Removes your right to go to court"),
    "auto renewal":               ("LOW",     5, "Can trap you in contract"),
    "no oral modification":       ("LOW",     3, "All changes must be written"),
}

def score_contract(text: str, role: str) -> dict:
    text_lower  = text.lower()
    total_score = 0
    found_flags = []

    for clause, (severity, weight, description) in RED_FLAG_CLAUSES.items():
        if clause in text_lower:
            # Find the actual sentence containing the clause
            sentences = re.split(r'(?<=[.!?])\s+', text)
            excerpt   = next(
                (s[:200] for s in sentences if clause in s.lower()),
                "Clause detected but excerpt unavailable"
            )
            total_score += weight
            found_flags.append({
                "clause":      clause.title(),
                "severity":    severity,
                "weight":      weight,
                "description": description,
                "excerpt":     excerpt,
            })

    # Sort by severity weight
    found_flags.sort(key=lambda x: x["weight"], reverse=True)
    final_score = min(95, total_score)

    return {
        "score": final_score,
        "flags": found_flags,
        "grade": "HIGH" if final_score > 65 else "MODERATE" if final_score > 35 else "LOW",
        "flag_count": len(found_flags),

import io

def generate_notice_pdf(notice_text: str, metadata: dict) -> bytes:
    buffer = io.BytesIO()
    doc    = SimpleDocTemplate(
        buffer, pagesize=A4,
        rightMargin=2.5*cm, leftMargin=2.5*cm,
        topMargin=2.5*cm,   bottomMargin=2*cm,
    )

    styles = getSampleStyleSheet()
    gold   = colors.HexColor("#A87828")
    navy   = colors.HexColor("#1B2236")

    title_style = ParagraphStyle(
        "Title", parent=styles["Heading1"],
        fontSize=18, textColor=navy,
        fontName="Times-Bold", spaceAfter=6,
    )
    body_style = ParagraphStyle(
        "Body", parent=styles["Normal"],
        fontSize=10.5, leading=16,
        textColor=colors.HexColor("#333333"),
    )
    footer_style = ParagraphStyle(
        "Footer", parent=styles["Normal"],
        fontSize=8, textColor=colors.grey, alignment=1,
    )

    story = [
        Paragraph("LEGAL NOTICE", title_style),
        Paragraph(f"Prepared by AIttorney · {metadata.get('date','')}", footer_style),
        Spacer(1, 0.3*cm),
        HRFlowable(width="100%", thickness=1, color=gold),
        Spacer(1, 0.4*cm),
    ]

    for line in notice_text.split("\n"):
        if line.strip():
            story.append(Paragraph(line, body_style))
            story.append(Spacer(1, 0.15*cm))

    story += [
        Spacer(1, 0.5*cm),
        HRFlowable(width="100%", thickness=0.5, color=colors.lightgrey),
        Spacer(1, 0.2*cm),
        Paragraph(
            "This document was generated by AIttorney for educational purposes only. "
            "It does not constitute legal advice. Please have this reviewed by a licensed advocate before use.",
            footer_style,
        ),
    ]

    doc.build(story)
    return buffer.getvalue()Now you can say "generates professionally formatted PDF output" — that's a real deliverable.PHASE 2 — Add Real Technical Depth (Week 2)

EMBED_MODEL = SentenceTransformer("all-MiniLM-L6-v2")
client      = chromadb.Client()

def get_or_create_collection(name: str):
    return client.get_or_create_collection(name)

def embed_contract(text: str, doc_id: str):
    """Chunk contract into 500-char overlapping segments and embed."""
    collection = get_or_create_collection("contracts")
    chunks     = _chunk_text(text, chunk_size=500, overlap=50)

    embeddings = EMBED_MODEL.encode(chunks).tolist()
    ids        = [f"{doc_id}_chunk_{i}" for i in range(len(chunks))]
    metadatas  = [{"chunk_index": i, "doc_id": doc_id} for i in range(len(chunks))]

    collection.add(
        documents=ids,
        embeddings=embeddings,
        ids=ids,
        metadatas=metadatas,
    )
    # Store actual text separately
    return dict(zip(ids, chunks))

def semantic_search(query: str, doc_id: str, top_k: int = 3) -> list[str]:
    """Find the most relevant chunks for a question."""
    collection   = get_or_create_collection("contracts")
    query_embed  = EMBED_MODEL.encode([query]).tolist()

    results = collection.query(
        query_embeddings=query_embed,
        n_results=top_k,
        where={"doc_id": doc_id},
    )
    return results["documents"][0] if results["documents"] else []

def _chunk_text(text: str, chunk_size: int, overlap: int) -> list[str]:
    chunks, i = [], 0
    while i < len(text):
        chunks.append(text[i:i + chunk_size])
        i += chunk_size - overlap
    return chunksNow your contract chat uses actual semantic retrieval — it finds the 3 most relevant clauses before answering, not just "here's the entire document." def save_case_search(username: str, query: str, result: str, win_prob: int):
    conn = sqlite3.connect(DB_PATH)
    conn.execute(
        "INSERT INTO case_searches (username, query, result, win_prob) VALUES (?,?,?,?)",
        (username, query[:500], result[:3000], win_prob)
    )
    conn.commit()
    conn.close()

def get_case_history(username: str) -> list[dict]:
    conn  = sqlite3.connect(DB_PATH)
    rows  = conn.execute(
        "SELECT query, win_prob, created_at FROM case_searches WHERE username=? ORDER BY created_at DESC LIMIT 10",
        (username,)
    ).fetchall()
    conn.close()
    return [{"query": r[0], "win_prob": r[1], "date": r[2]} for r in rows]
